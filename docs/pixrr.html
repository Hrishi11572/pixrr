 92        # 1. Generate Pascal's Triangle row for smoothing

 95            for k in range(n):
 96                row.append(row[k] * (n - k) // (k + 1))

100        s = get_pascal_row(size - 1)

102        # Derivative vector (d) 
103        # Logic: difference of the Pascal row one degree smaller
104        d_prev = get_pascal_row(size - 2)

106        d[:-1] += d_prev
107        d[1:] -= d_prev
108
109        # 2. Create Kernels using Outer Product
110        # Gx (Horizontal) detects vertical edges
111        sobel_x = np.outer(s, d)
112        
113        # Gy (Vertical) detects horizontal edges
114        sobel_y = np.outer(d, s)
115        
116        return sobel_x, sobel_y

119

121        horizontal_gradient = conv2D(img, sobel_horizontal, hstep , vstep)
122        return horizontal_gradient

10    '''
11    docstring for padd_image: 
12    
13    :param img : input image to be padded as a numpy array 
14    :type img : np.ndarray 
15    :param row_pad: the number of rows to padd (default is 1)

27            padded[i + row_pad, j + col_pad] = img[i, j]
28    return padded

44    '''
45    Docstring for otsu_thresholding
46

56    def otsu_intraclass_variance(img: np.ndarray , threshold: int):
57        ''' https://en.wikipedia.org/wiki/Otsu%27s_method ''' 
58        return np.nansum(
59            [
60                np.mean(cls) * np.var(img, where=cls)
61                #   weight   ·  intra-class variance
62                for cls in [img >= threshold, img < threshold]
63            ]
64        )

67        range(np.min(img) + 1, np.max(img)),
68        key=lambda th: otsu_intraclass_variance(img, th),
69    )
